from PIL import Image


def binary_threshold(image_path, output_path, threshold=128):
    """
    对图片进行二值化处理并保存为新图片。

    参数:
    image_path (str): 输入图片的路径。
    output_path (str): 输出图片的路径。
    threshold (int): 二值化的阈值，默认为128。
    """
    try:
        # 打开图片
        with Image.open(image_path) as img:
            # 转换为灰度图
            img_gray = img.convert("L")
            # 应用阈值进行二值化
            img_binary = img_gray.point(lambda x: 255 if x > threshold else 0, "1")
            # 保存二值化后的图片
            img_binary.save(output_path)
            print(f"Binary image saved as {output_path}")
    except IOError as e:
        print(f"Error processing image: {e}")


if __name__ == "__main__":
    image_path = "key.png"  # 输入图片路径
    output_path = "new.png"  # 输出图片路径
    binary_threshold(image_path, output_path)
